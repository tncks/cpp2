String::String(const char *str)
	:str(0), len(0)
{
	for (len = 0; str[len]; len++);
	this->str = new char[len + 1];
	for (unsigned int i = 0; i < len + 1; i++)
	{
		this->str[i] = str[i];
	}
}

String::~String()
{
	delete[] str;
	len = 0;
}

char &String::Get(int index)
{
	return str[index];
}

const char &String::Get(int index) const
{
	return str[index];
}
